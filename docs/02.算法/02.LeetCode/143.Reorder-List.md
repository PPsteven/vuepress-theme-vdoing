## 代码

```python
class Solution:
    def reorderList(self, head: ListNode) -> None:
        def findMiddle(head):
            slow, fast = head, head.next
            while fast and fast.next:
                slow = slow.next 
                fast = fast.next.next 
            mid = slow.next 
            slow.next = None 
            return mid 
        def reverse(head):
            pre, cur = None, head
            while cur:
                cur_next_bak = cur.next 
                cur.next = pre 
                pre = cur
                cur = cur_next_bak 
            return pre
        
        def mergeTwoList(l1, l2):
            dummyHead = ListNode(None)
            p = dummyHead
            while l1 and l2:
                p.next = l1
                p = p.next 
                l1 = l1.next 
                p.next = l2
                p = p.next 
                l2 = l2.next 
            p.next = l1 if l1 else l2 
            return dummyHead.next 
                 
        if not head or not head.next: return head 
        l1, l2 = head, findMiddle(head)
        l2 = reverse(l2)
        return mergeTwoList(l1, l2)
```
