```python
class Solution:
    def reverseBetween(self, head: ListNode, m: int, n: int) -> ListNode:
        pre, cur = None, head 
        for _ in range(m - 1):
            pre = cur 
            cur = cur.next
        before, after = pre, cur 

        for _ in range(n - m + 1):
            cur_next_bak = cur.next
            cur.next = pre 

            pre = cur 
            cur = cur_next_bak
        if before:
            before.next = pre 
        else:
            head = pre 
        after.next = cur 
        return head 
```
